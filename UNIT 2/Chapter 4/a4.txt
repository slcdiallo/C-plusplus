#include <iostream>

using namespace std;

int main()
{
string categories[]={"BREAD", "CHEESE", "MEAT", "VEGGIE", "SIDE"};
string options[5][6]={
{"Wheat", "Sourdough", "Italian", "White", "Multigrain","NONE"},
{"Mozzarella", "Cheddar", "Swiss", "Provolone", "Monterey Jack","Goat Cheese"},
{"Chicken", "Turkey", "Roast Beef","NONE","NONE"},
{"Mushrooms", "Olives", "Spinach", "Peppers", "Onion","NONE"},
{"Fries", "Chips", "Salad", "Coleslaw","Tomato Soup","NONE"}
};
int total = sizeof(categories)/sizeof(categories[0]);
string input[5];
int num=0;
for(int i=0;i<total;i++){
cout<<"Enter your choice for "<<categories[i]<<" (";
int n = 6;
for(int j=0;j<n;j++){
if(j!=n-1 && options[i][j].compare("NONE")!=0){
cout<<j+1<<" for "<<options[i][j];
cout<<", ";
}
else{
cout<<0<<" for NONE"<<" ): ";
break;
}
}
cin>>num;
if(num==0){
input[i]="NONE";
}
else{
input[i]=options[i][num-1];
}
cout<<"The "<<categories[i]<<" name selection is "<<input[i]<<endl;
}
cout<<"You ordered a Grilled Cheese on "<<input[0]<<" bread with "
<<input[1]<<", "<<input[2]<<", "
<<input[3]<<", "<<input[4]<<"."<<endl;
double price=5; //base price
if(input[2].compare("NONE")!=0){ //meat is none
price+=2;
}
if(input[3].compare("NONE")!=0){ // no veggies
price+=1;
}
if(input[4].compare("NONE")!=0){ // no sides
price+=2;
}
cout<<"Your balance is: $"<<price;
  
return 0;
}
